KFFNDRELDETS1	Keyword Framework Macro		0	N	Y	A
00001	DECLARE	s	VALUATION,OPTION,RELEASE_PAYMENT,AUTH_THRESHOLD,TIERENABLED	
00002	IF		$APPTYPE = 'F',REVAL	
00003	LET		VALUATION = 'N'	
00004	SELECT		''Y' INTO :VALUATION FROM BIT08 WHERE BIT08.MORTGAGE = :$MORTGAGE;'	
00005	IF		VALUATION = 'Y',REVAL	
00006	WINDOW		'W_AP010_39','',''	Valuation Report
00007	JUMP		NEXT	
00008	WINDOW	REVAL	'W_AP010_39','ENQUIRY',''	Valuation Report
00009	WINDOW		'W_AP010_38','',''	Re-Valuation Report
00010	JUMP		NEXT	
00011	SERVICE	NEXT	'NVUO_MACRO_NEWSTAGE','A','50'	Check Whether Case Can Move To New Stage
00012	SELECT		'CVC.SAM05.TIER INTO :TIERENABLED FROM CVC.SAM05, CVC.ACF94 WHERE CVC.ACF94.MORTGAGE = :$MORTGAGE AND CVC.ACF94.MTG_SOC_SEQNO = CVC.SAM05.SOCSEQNO'	
00013	IF		TIERENABLED # 'Y',NEXT3	
00014	SERVICE		'NVUO_MACRO_TIER_REFRESH','',''	Application Tier Refresh
00015	SERVICE		'NVUO_MACRO_TIER_CHECK','',''	Application Tier Check
00016	IF	NEXT3	$RETURNDATA = 'Y',END	
00017	SERVICE		'NVUO_MACRO_RETENTION_APPORTION','',''	Retention Apportionment
00018	WINDOW		'W_AP034_01','',''	Maintain Completion Details
00019	SERVICE		'NVUO_MACRO_PRE_COMPLETE','',''	Generate Funds Release Records
00020	WINDOW		'W_AP038_01','',''	Maintain Funds Release Details
00021	SELECT		'NVL(SAM05.FNDREL_AUTH_THRESHOLD,0) INTO :AUTH_THRESHOLD FROM CVC.SAM05, CVC.ACF94 WHERE SAM05.SOCSEQNO = ACF94.MTG_SOC_SEQNO AND ACF94.MORTGAGE = :$MORTGAGE'	
00022	SELECT		'ACT18.RELPAYAMT INTO :RELEASE_PAYMENT FROM ACT18 WHERE ACT18.MORTGAGE = :$MORTGAGE AND ACT18.APPLSEQ = :$APPLSEQ'	
00023	IF		AUTH_THRESHOLD = 0,SIGNOFF	
00024	IF		RELEASE_PAYMENT > AUTH_THRESHOLD,CONT	
00025	SERVICE	SIGNOFF	'NVUO_MACRO_MANAGER_SIGNOFF','',''	Funds Release Final Manager Signoff
00026	SERVICE		'S_REPAYMENTAPR','',''	Repayment/Apr Calculation
00027	MESSAGEBOX		OPTION,'Fund Release','Funds Will Automatically be Released',INFORMATION,OK	
00028	JUMP	END	NEXT2	Repayment/Apr Calculation
00029	MESSAGEBOX	CONT	OPTION,'Funds Release','Funds Will Automatically be Released',INFORMATION,OK	
00030	COMMENT	NEXT2	Execute KFFNDRELDETS2 and KFFNDRELDETS3 subsequently after executing this macro.	End of Macro
